76  virtual void work() override {
77 
78  invisibles->clear();
79  jacobians->clear();
80 
81  // Equations to solve:
82  //(1) (p1 + p2)^2 = s12 = M1^2 + M2^2 + 2E1E2 + 2p1xp2x + 2p1yp2y + p1zp2z
83  //(2) p1x = - pTx #Coming from pT neutrino = -pT visible = - (p2 + ISR)
84  //(3) p1y = - pTy #Coming from pT neutrino = -pT visible = - (p2 + ISR)
85  //(4) M1 = 0 -> E1^2 = p1x^2 + p1y^2 + p1z^2
86 
87  // Don't spend time on unphysical part of phase-space
88  if(*s12 > SQ(sqrt_s))
89  return;
90 
91  const LorentzVector& p2 = m_particle_tags[0].get<LorentzVector>();
92 
93  // FIXME
94  const LorentzVector ISR; // = (*particles)[0];
95  auto pT = p2 + ISR;
96 
97  const double p22 = p2.M2();
98 
99  // From eq.(1) p1z = -B*E1 + A
100  // From eq.(4) + eq.(1) (1-B^2)* E1^2 + 2AB* E1 - C = 0
101 
102  const double A = (*s12 - p22 + 2*(pT.Px()*p2.Px() + pT.Py()*p2.Py()))/(2*p2.Pz());
103  const double B = p2.E()/p2.Pz();
104  const double C = SQ(pT.Px()) + SQ(pT.Py());
105 
106  // Solve quadratic a*E1^2 + b*E1 + c = 0
107  const double a = 1 - SQ(B);
108  const double b = 2*A*B;
109  const double c = -C;
110 
111  std::vector<double> E1;
112 
113  solveQuadratic(a, b, c, E1, false);
114 
115  if (E1.size() == 0)
116  return;
117 
118  for(unsigned int i=0; i<E1.size(); i++){
119  const double e1 = E1.at(i);
120 
121  if (e1 < 0.) continue;
122 
123  LorentzVector p1(-pT.Px(), -pT.Py(), A - B*e1, e1);
124 
125  // Check if solutions are physical
126  LorentzVector tot = p1 + p2;
127  double q1Pz = std::abs(tot.Pz() + tot.E()) / 2.;
128  double q2Pz = std::abs(tot.Pz() - tot.E()) / 2.;
129  if(q1Pz > sqrt_s/2 || q2Pz > sqrt_s/2)
130  continue;
131 
132  invisibles->push_back({p1});
133  jacobians->push_back(computeJacobian(p1, p2));
134  }
135  }
