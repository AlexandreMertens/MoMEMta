48  int module_table_newindex(lua_State* L) {
49 
50  lua_getmetatable(L, 1);
51  lua_getfield(L, -1, "__type");
52 
53  const char* module_type = luaL_checkstring(L, -1);
54  const char* module_name = luaL_checkstring(L, 2);
55 
56  // Remove field name from stack
57  lua_pop(L, 1);
58 
59  lua_getfield(L, -1, "__ptr");
60  void* cfg_ptr = lua_touserdata(L, -1);
61  ConfigurationReader* reader = static_cast<ConfigurationReader*>(cfg_ptr);
62 
63  reader->addModule(module_type, module_name);
64 
65  // Remove metatable and field name from stack
66  lua_pop(L, 2);
67 
68  // And actually set the value to the table
69  lua_rawset(L, 1);
70 
71  return 0;
72  }

74  void lua_register_modules(lua_State* L, void* ptr) {
75  std::vector<std::string> modules = ModuleFactory::get().getPluginsList();
76  for (const auto& module: modules) {
77  const char* module_name = module.c_str();
78  char* module_metatable = new char[module.size() + 3 + 1];
79  strncpy(module_metatable, module_name, module.size() + 1);
80  strncat(module_metatable, "_mt", 3);
81 
82  int type = lua_getglobal(L, module_name);
83  lua_pop(L, 1);
84  if (type != LUA_TNIL) {
85  // Global already exists
86  continue;
87  }
88 
89  // Create a new empty table
90  lua_newtable(L);
91 
92  // Create the associated metatable
93  luaL_newmetatable(L, module_metatable);
94 
95  lua_pushstring(L, module_name);
96  lua_setfield(L, -2, "__type");
97 
98  lua_pushlightuserdata(L, ptr);
99  lua_setfield(L, -2, "__ptr");
100 
101  // Set the metadata '__newindex' function
102  const luaL_Reg l[] = {
103  {"__newindex", lua::module_table_newindex},
104  {nullptr, nullptr}
105  };
106  luaL_setfuncs(L, l, 0);
107 
108  lua_setmetatable(L, -2);
109 
110  // And register it as a global variable
111  lua_setglobal(L, module_name);
112 
113  TRACE("Registered new lua global variable '{}'", module_name);
114 
115  delete[] module_metatable;
116  }
117  }

138 ConfigurationReader::ConfigurationReader(const std::string& file) {
139 
140  LOG(debug) << "Parsing LUA configuration from " << file;
141  lua_state = luaL_newstate();
142 
143  luaL_openlibs(lua_state);
144 
145  // Register function load_modules
146  lua_pushlightuserdata(lua_state, this);
147  lua_pushcclosure(lua_state, lua::load_modules, 1);
148  lua_setglobal(lua_state, "load_modules");
149 
150  // Register existing modules
151  lua::lua_register_modules(lua_state, this);
152 
153  // Parse file
154  if (luaL_dofile(lua_state, file.c_str())) {
155  printf("%s\n", lua_tostring(lua_state, -1));
156  }
157 
158  // FIXME: Find a better way of doing that
159 
160  // Read global configuration from global variable named 'configuration'
161  m_global_configuration.reset(new ConfigurationSet("configuration", "configuration"));
162  int type = lua_getglobal(lua_state, "configuration");
163  if (type == LUA_TTABLE) {
164  LOG(debug) << "Parsing global configuration.";
165  m_global_configuration->parse(lua_state, -1);
166  }
167  lua_pop(lua_state, 1);
168 
169  // Read vegas configuration
170  m_vegas_configuration.reset(new ConfigurationSet("vegas", "vegas"));
171  type = lua_getglobal(lua_state, "vegas");
172  if (type == LUA_TTABLE) {
173  LOG(debug) << "Parsing vegas configuration.";
174  m_vegas_configuration->parse(lua_state, -1);
175  }
176  lua_pop(lua_state, 1);
177 
178  for (auto& m: m_light_modules) {
179  LOG(debug) << "Configuration declared module " << m.type << "::" << m.name;
180 
181  lua_getglobal(lua_state, m.type.c_str());
182  lua_getfield(lua_state, -1, m.name.c_str());
183 
184  m.parameters.reset(new ConfigurationSet(m.type, m.name, m_global_configuration));
185  m.parameters->parse(lua_state, -1);
186 
187  lua_pop(lua_state, 2);
188  }
189 }
