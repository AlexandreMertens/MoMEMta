104  template<typename T> std::shared_ptr<T> put(const InputTag& tag) {
105  auto it = m_storage.find(tag);
106  if (it != m_storage.end()) {
107  if (it->second.valid)
108  throw duplicated_tag_error("A module already produced the tag '" + tag.toString() + "'");
109  // A module already requested this block in read-mode. Since the memory is allocated, simply consider the block as valid
110  it->second.valid = true;
111 
112  // If the block is empty, it's a delayed instanciation. Simply flag the block as valid, and allocate memory for it
113  if (it->second.ptr.empty()) {
114  std::shared_ptr<T> ptr(new T());
115  it->second.ptr = boost::any(ptr);
116  }
117 
118  } else {
119  it = create<T>(tag, true);
120  }
121 
122  // Update current module description
123  assert(! m_current_module.empty());
124  Description& description = m_description[m_current_module];
125  description.outputs.push_back(tag.parameter);
126 
127  return boost::any_cast<std::shared_ptr<T>>(it->second.ptr);
128  }
