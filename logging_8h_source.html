29 class colored_formatter: public spdlog::formatter

31  void format(spdlog::details::log_msg& msg) override
32  {
33 #ifndef SPDLOG_NO_DATETIME
34  auto tm_time = spdlog::details::os::localtime(spdlog::log_clock::to_time_t(msg.time));
35  auto duration = msg.time.time_since_epoch();
36  auto millis = std::chrono::duration_cast<std::chrono::milliseconds>(duration).count() % 1000;
37 
38  msg.formatted << '[' << static_cast<unsigned int>(tm_time.tm_year + 1900) << '-'
39  << fmt::pad(static_cast<unsigned int>(tm_time.tm_mon + 1), 2, '0') << '-'
40  << fmt::pad(static_cast<unsigned int>(tm_time.tm_mday), 2, '0') << ' '
41  << fmt::pad(static_cast<unsigned int>(tm_time.tm_hour), 2, '0') << ':'
42  << fmt::pad(static_cast<unsigned int>(tm_time.tm_min), 2, '0') << ':'
43  << fmt::pad(static_cast<unsigned int>(tm_time.tm_sec), 2, '0') << '.'
44  << fmt::pad(static_cast<unsigned int>(millis), 3, '0') << "] ";
45 
46 //no datetime needed
47 #else
48  (void)tm_time;
49 #endif
50 
51 #ifndef SPDLOG_NO_NAME
52  msg.formatted << '[' << msg.logger_name << "] ";
53 #endif
54 
55  msg.formatted << '[';
56  switch (msg.level) {
57  case spdlog::level::trace:
58  case spdlog::level::debug:
59  msg.formatted << color(90);
60  break;
61  case spdlog::level::warn:
62  msg.formatted << color(33);
63  break;
64  case spdlog::level::err:
65  case spdlog::level::critical:
66  case spdlog::level::alert:
67  case spdlog::level::emerg:
68  msg.formatted << color(31);
69  break;
70  default:
71  break;
72  }
73  msg.formatted << spdlog::level::to_str(msg.level) << color(0) << "] ";
74  msg.formatted << fmt::StringRef(msg.raw.data(), msg.raw.size());
75  msg.formatted << spdlog::details::os::eol();
76  }
