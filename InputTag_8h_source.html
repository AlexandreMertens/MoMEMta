55  static bool isInputTag(const std::string& tag) {
56  if (tag.find("::") == std::string::npos)
57  return false;
58 
59  if (tag.find("/") != std::string::npos) {
60  auto tags = split(tag, "::");
61  auto rtags = split(tags[1], "/");
62  try {
63  size_t index = std::stoull(rtags[1]);
64  UNUSED(index);
65  return true;
66  } catch (std::invalid_argument e) {
67  return false;
68  }
69  } else {
70  return true;
71  }
72  }
